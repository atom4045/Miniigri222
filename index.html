-- Добрый скрипт для веселья (Roblox) local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/xHeptc/Kavo-UI-Library/main/source.lua"))() local Window = Library.CreateLib("ВЕСЕЛЫЙ СКРИПТ", "DarkTheme") -- Вкладка "Главная" local Main = Window:NewTab("Функции") local MainSection = Main:NewSection("Движение и Тапы") -- 1. Бесконечный прыжок (Auto-Jump) local autoJumpEnabled = false MainSection:NewToggle("Авто-Прыжок", "Ты будешь прыгать без остановки", function(state) autoJumpEnabled = state game:GetService("RunService").RenderStepped:Connect(function() if autoJumpEnabled then if game.Players.LocalPlayer.Character and game.Players.LocalPlayer.Character:FindFirstChild("Humanoid") then game.Players.LocalPlayer.Character.Humanoid.Jump = true end end end) end) -- 2. Супер Скорость MainSection:NewSlider("Скорость бега", "Меняет твою скорость", 250, 16, function(s) game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = s end) -- 3. Авто-Кликер (для симуляторов) local autoClickEnabled = false MainSection:NewToggle("Авто-Кликер", "Сам жмет на экран", function(state) autoClickEnabled = state spawn(function() while autoClickEnabled do local VirtualUser = game:GetService("VirtualUser") VirtualUser:CaptureController() VirtualUser:ClickButton1(Vector2.new(0,0)) wait(0.1) -- Скорость клика (чем меньше число, тем быстрее) end end) end) -- Вкладка "Инфо" local Info = Window:NewTab("Инфо") local InfoSection = Info:NewSection("Скрипт создан для веселья!") InfoSection:NewButton("Удалить меню", "Закрыть чит", function() game:GetService("CoreGui"):FindFirstChild("ВЕСЕЛЫЙ СКРИПТ"):Destroy() end) print("Скрипт успешно запущен в Delta!")
